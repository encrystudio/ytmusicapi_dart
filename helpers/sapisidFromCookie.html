String sapisidFromCookie(String rawCookie) {
  final cookies = <String, String>{};
  for (final pair in rawCookie.replaceAll('"', '').split(';')) {
    final kv = pair.split('=');
    if (kv.length == 2) cookies[kv[0].trim()] = kv[1].trim();
  }
  return cookies['__Secure-3PAPISID'] ?? '';
}
