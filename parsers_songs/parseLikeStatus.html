String parseLikeStatus(JsonMap service) {
  final status = ['LIKE', 'INDIFFERENT'];
  final idx =
      status.indexOf((service['likeEndpoint'] as JsonMap)['status'] as String) -
      1;
  return status[idx < 0 ? 0 : idx];
}
